[1] Direct function of the [Hiding the Parent Element] by clicking on the Button :
var myMainElem = document.getElementById('main').parentElement,

myBtn = myMainElem.lastElement;

myBtn.onclick = function() {
myMainElem.style.display = 'none'

}

[2] Event function of the [Hiding the Parent Element] by When the {Window Load} :
var myMainElem = document.getElementById('main').parentElement,

myBtn = myMainElem.lastElement;

myBtn.onclick = function() {
myMainElem.style.display = 'none'

}

window.onload = function() {
myBtn.click();

}

[3] Event function of the [Hiding the Parent Element] by When the {After 2 of Window Load} + {setTimeout()} :
var myMainElem = document.getElementById('main').parentElement,

myMainElem.style.display = 'none'

}

window.onload = function() {
'use strict';

function() {
myBtn.click();
