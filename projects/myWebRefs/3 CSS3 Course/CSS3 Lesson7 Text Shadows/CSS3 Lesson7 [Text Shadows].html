[ Text Shadows ] Syntax & Methods:

// *) [Text Shadows] Syntax as following :

//1) {Exmple 1} [image Element with Hovering(box-shadow) + Transition] Effect Syntax:
//a) [HTML] file :
< img src='' alt=''>
// ------------------------------------------------------------------------------

//b) [CSS3] file :
img {
margin: 40px;
padding: 3px;
border: 1px solid #CCC;
backgournd-color: #FFF;
border-radius: 50%;
transition: .5s ease-in-out; => {this will make hovering Effect displayed in Transition Mode}
}
// ------------------------------------------------------------------------------

img: hover {
box-shadow :7px 7px 20px #E4E4E4; => {This will make a circle shape}
}
// ------------------------------------------------------------------------------

// ------------------------------------------------------------------------------

//2) {Exmple 2} [image Element with Hovering(border-radius) + Transition] Effect Syntax:
//a) [HTML] file :
< img src='' alt=''>
// ------------------------------------------------------------------------------

//b) [CSS3] file :
img {
margin: 40px;
padding: 3px;
border: 1px solid #CCC;
box-shadow :7px 7px 20px #E4E4E4;
backgournd-color: #FFF;
transition: .5s ease-in-out; => {this will make hovering Effect displayed in Transition Mode}
}
// ------------------------------------------------------------------------------

img: hover {
border-radius: 50%;
}
// ------------------------------------------------------------------------------
