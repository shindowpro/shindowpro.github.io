[Javascript Array Search] Concept:

=> [Array Search Function] : Is one of the most important controling Function of Array in JS programing language as a concept which allow to Search About the Index of specific Element in Array By it's Value regardeless of the Data type of Array, using different Methods.

=> [Array Search Function] : Array Searching Function is Getting the Index of Element Value Array's Element Based on (Zero-based) Using Two Main Method ( Start - End ) or (End - Start) by Basic & Advanced Searching Function .

[Javascript Array Searching] Importance & Benefits:

=> [Array Search] : Is Essential to any programmer or developer to provide more Control of the Array by providing the Ability to Find or Get Any Element's Index of any Array Type & make any Programable Process.

=> [Array Search] : Is Not only Essential to perform and processing array's Element, but Also even with any Element in the Web Page , using [HTML] , [CSS] programing Languages whether this Element(s) was/were Word, Letter , Paragraph, Number of letters, and pursing speicific action(s) or Function(s)

[Javascript Array Search] Mains Methods:

=>[Array Search Functions (Methods & syntaxs] are as following :

  1. => Array Search (Start - End) Method {ArrName.indexOf("Element Value");} ]:

    2. => [Search Function (Start - End) Method]: Is the most common method usually used to Search / Find specific Element Index Array in Js P.language, by the Acsecnding Arrangement of Array Index [0 - Last Array Index]

    =>[Search (Start - End) Method] Syntax as following:

    // Basic Search Function (Basic Searching on Element1 Index) :
    var arrayName = ["Element1", "Element2", "Element3"];
       arrayItem = arrayName.indexOf("Element1 Value");
    document.getElementById("HTMLTagName").innerHTML =
       "Index of Element Value :" + arrayItem ;

    // Advanced Search Function (Advanced Searching on Element1 Index):
    var arrayName = ["Element1", "Element2", "Element3"];
       arrayItem = arrayName.indexOf("Element1 Value");
    document.getElementById("HTMLTagName").innerHTML =
       "Index of Element Value :" + arrayItem ;

  2. => Array Search (End - Start) Method {ArrName.lastIndexOf("Element Value");} ]

    3. => [Search Function (End - Start)]: Is the most common method usually used to Search / Find specific Element Index Array in Js P.language, by the Descengly Arrangement of Array Index [Last Array Index - 0]

    =>[Search (End - Start) Method] Syntax as following:

    // Basic Search Function (Basic Searching on Element1 Index) :
    var arrayName = ["Element1", "Element2", "Element3"];
       arrayItem = arrayName.lastIndexOf("Element1 Value");
    document.getElementById("HTMLTagName").innerHTML =
       "Index of Element Value :" + arrayItem ;

    // Advanced Search Function (Advanced Searching on Element1 Index):
    var arrayName = ["Element1", "Element2", "Element3"];
       arrayItem = arrayName.lastIndexOf("Element1 Value");
    document.getElementById("HTMLTagName").innerHTML =
       "Index of Element Value :" + arrayItem ;

    /* global alert, document, prompt */
    /* Array Search Methods [Start - End] & [End - Start] Application : */ :
    function Search() {
    var pass = prompt("Press (1) to Start the Appliction \n Press any other key to Exit ...");
       if (pass == 1) {
         alert("Welcome to Array Searching Application ...");
         var choice = prompt(" Press (1) to Choose [Start - End] Method \n Press (2) to Choose [End - Start] Method");
         switch (choice) {
           case 1 :
             alert("Welcome to [Start - End] Array Search Method ...\n Lets Start Inserting your Array...");
             var count = parseInt(prompt("Enter The Count of Your Array :")),
               arr = new Array (count);
             for (i = 0; i < count; i++ ) {
               arr[i] = prompt("Enter your Element number (" + (i + 1) + ")");
             }
             var InternalM = prompt("Press (1) for Basic Searching Function ... \n
               Press (2) for Advanced Searching Function ");
             switch (InternalM) {
               case 1 :
                 alert("Welcome to [Start - End] / (Basic) Search Method ...");
                 var ElV = prompt("Enter your Element Value your want to search about it's Index ... " ),
                   SV = arr.indexOf(ElV);
                 document.getElementById("DisplayMainArr").innerHTML =
                   "Your Main Array : " + arr;
                   document.getElementById("DisplaySearchBas").innerHTML =
                     "Your Element ( " + ElV + " ) Index is : " + "<br>" + SV;
               break;
               case 2 :
                 alert("Welcome to [Start - End] / (Advanced) Search Method ...");
                 var ElV = prompt("Enter your Element Value your want to search about it's Index ... " ),
                   StartIndex = parseInt(prompt("Enter your Search Start Postion : ")) - 1,
                   SV = arr.IndexOf(ElV, StartIndex);
                 document.getElementById("DisplayMainArr").innerHTML =
                   "Your Main Array : " + arr;
                 document.getElementById("DisplaySearchAdv").innerHTML =
                   "Your Element ( " + ElV + " ) Index is : " + "<br>" + SV;
                 break;
               }            default :
                 alert("Invalid Entry!");
                 break;
           case 2 :
             alert("Welcome to [Start - End] Array Search Method ...\n
               Lets Start Inserting your Array...");
             var count = parseInt(prompt("Enter The Count of Your Array :")),
             arr = new Array (count);
             for (i = 0; i < count; i++ ) {
               arr[i] = prompt("Enter your Element number (" + (i + 1) + ")");
             }
             var InternalM = prompt("Press (1) for Basic Searching Function ... \n
               Press (2) for Advanced Searching Function ");
             switch (InternalM) {
               case 1 :
                 alert("Welcome to [End - Start] / (Basic) Search Method ...");
                 var ElV = prompt("Enter your Element Value your want to search about it's Index ... " ),
                 SV = arr.lastIndexOf(ElV);
                 document.getElementById("DisplayMainArr").innerHTML =
                   "Your Main Array : " + arr;
                 document.getElementById("DisplaySearchBas").innerHTML =
                   "Your Element ( " + ElV + " ) Index is : " + "<br>" + SV;
                 break;
               case 2 :
                 alert("Welcome to [End - Start] / (Advanced) Search Method ...");
                 var ElV = prompt("Enter your Element Value your want to search about it's Index ... " ),
                   StartIndex = parseInt(prompt("Enter your Search Start Postion : ")) - 1,
                   SV = arr.lastIndexOf(ElV, StartIndex);
                 document.getElementById("DisplayMainArr").innerHTML =
                   "Your Main Array : " + arr;
                 document.getElementById("DisplaySearchAdv").innerHTML =
                   "Your Element ( " + ElV + " ) Index is : " + "<br>" + SV;
              break; } break; } } else { alert("Good Bye ! "); } }
