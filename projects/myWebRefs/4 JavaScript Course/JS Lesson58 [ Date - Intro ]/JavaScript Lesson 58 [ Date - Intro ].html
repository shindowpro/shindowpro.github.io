=> JavaScript [ Date - Intro ] : Main Built-In Functions in JS to deal with Date and Time as following :

  1. new Date(); => For Full Format of Current Date & Time
  2. new Date(Number of MillSeconds); => For Full Format of Date & Time Starting from the Default [Unix Standards UTC (Thurs 1 Jan 1970 00:00:00 ) + Given Milliseconds & Defferance of Time Zone from GMT]
  3. new Date(String Format); => For Full Format of Current Date & Time by Certain String Customized Date Format
  4. new Date(7 Integer Arguments of Date & Time); => For Full Format of Current Date & Time by Certain Integer Customized Date Format

       Press (1) Full Format of Current Date & Time\n
       Press (2) Full Format of Date & Time After Adding Milliseconds to [Default Unix Standards UTC \n
       Press (3) Full Format of Current Date & Time by Certain String Customized Date Format \n

     if (meth !== "1" && meth !== "2" && meth !== "3" && meth !== "4") {
       alert("Good Bye!");
     } else {

     if (meth == "1") {
       alert("Welcome to Date & Time Format Built-In Method number (1) [Full Format of

       var myDate = new Date();

       document.getElementById("txtarea").innerHTML = "Your Date Format by [Full Format of

       alert("Welcome to Date & Time Format Built-In Method number (2) [Full Format of Date & Time
         After Adding Milliseconds to Default Unix Time Standard] ");

       var milS = parseInt(prompt("Enter your MilliSecond Number to Add to Default Time of Unix"));
         myDate = new Date(milS);

       alert("Your Date Format by [Full Format of Date & Time After Adding Milliseconds to
         Unix Time Standard]\n" + myDate);

         "Your Date Format by [Full Format of Date & Time After Adding Milliseconds to Default Unix Time Standard]\n" +
         myDate;
